   ```python
   class Madrasah(models.Model):
       ...
       # Enrollment Capacity
       total_capacity = PositiveIntegerField(
           default=0,
           help_text="Maximum student capacity"
       )
       current_enrollment = PositiveIntegerField(
           default=0,
           help_text="Current enrolled students"
       )

       # Operational Status
       class OperationalStatus(models.TextChoices):
           ACTIVE = 'active', 'Active'
           INACTIVE = 'inactive', 'Inactive'
           UNDER_CONSTRUCTION = 'construction', 'Under Construction'
           TEMPORARILY_CLOSED = 'temp_closed', 'Temporarily Closed'
           CLOSED = 'closed', 'Permanently Closed'

       operational_status = CharField(
           max_length=20,
           choices=OperationalStatus.choices,
           default='active'
       )

       def capacity_percentage(self):
           if self.total_capacity == 0:
               return 0
           return (self.current_enrollment / self.total_capacity) * 100
   ```
